## 素数相关

 埃氏筛
 按顺序第一次遇到的一定是素数.
 ```cpp
 class Solution {
 public:
     int countPrimes(int n) {
         vector<int>isPrime(n,1);
         int count=0;
         for(int i=2;i<n;i++){
             if(isPrime[i]==1){
                 count++;
                 for(long long ord=(long long)i*i;ord<n;ord+=i)
                 {
                     isPrime[ord]=0;
                 }
             }
         }
         return count;
     }
 };
 ```
